6   package zeroth.actor.screen.iface.web.rest;
7   import javax.ws.rs.WebApplicationException;
8   import javax.ws.rs.core.HttpHeaders;
9   import javax.ws.rs.core.Response;
10  import javax.ws.rs.core.Response.Status;
11  /**
12   * Throw this exception to return a 401 Unauthorized response.
13   * @author nilcy
14   */
15  public class UnauthorizedException extends WebApplicationException {
16      /** 製品番号 */
17      private static final long serialVersionUID = -6796772574451059099L;
18      /** コンストラクタ */
19      public UnauthorizedException() {
20          this("Please authenticate.", "Name of your web service");
21      }
22      /**
23       * コンストラクタ
24       * @param message message
25       * @param realm realm
26       */
27      public UnauthorizedException(final String message, final String realm) {
28          super(Response.status(Status.UNAUTHORIZED)
29              .header(HttpHeaders.WWW_AUTHENTICATE, "Basic realm=\"" + realm + "\"").entity(message)
30              .build());
31      }
32  }
