12  import zeroth.actor.service.app.actor.CustomerApplication;
13  import zeroth.actor.service.domain.Customer;

15  import zeroth.framework.enterprise.app.SimpleRepositoryApplication;
16  import zeroth.framework.screen.iface.jsf.AbstractActionImpl;

19  /**
20   * 得意先アクション
21   * @author nilcy
22   */
23  @Named(value = "customerAction")
24  @ConversationScoped
25  public class CustomerAction extends AbstractActionImpl<Customer, Long, SimpleFilter> {
26      /** 製品番号 */
27      private static final long serialVersionUID = 873776474936603723L;
28      /** 得意先アプリケーションI/F */
29      @Inject
30      @CustomerApplicationCDI
31      private CustomerApplication customerApplication;
32      /** コンストラクタ */
33      public CustomerAction() {
34      }
35      /** {@inheritDoc} */
36      @Override
37      public SimpleRepositoryApplication<Customer, Long, SimpleFilter> getApplication() {
38          return customerApplication;
39      }
40      /** {@inheritDoc} */
41      @Override
42      protected Customer createInstance() {
43          return new CustomerFactory().create();
44      }
45      /** {@inheritDoc} */
46      @Override
47      protected boolean beforeSave() {
48          final Customer c = getSelected();
49          c.getAddress().setName(
50              StringUtils.defaultIfEmpty(c.getAddress().getName(), c.getOfficialName()));
51          c.getContact().setName(
52              StringUtils.defaultIfEmpty(c.getContact().getName(), c.getOfficialName()));
53          return super.beforeSave();
54      }
55      /** {@inheritDoc} */
56      @Override
57      protected SimpleFilter createFilter() {
58          return FilterFactory.createSimpleFilter();
59      }
60  }
