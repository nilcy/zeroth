6   package zeroth.actor.service.domain.parts;
7   import java.util.Arrays;
8   /**
9    * 権限
10   * @author nilcy
11   */
12  public enum Auth {
13      /** 監督者 */
14      SUPERVISOR,
15      /** 管理者 */
16      ADMINISTRATOR,
17      /** 所属長 */
18      MANAGER,
19      /** 推進者 */
20      LEADER,
21      /** 担当者 */
22      MEMBER,
23      /** 支援者 */
24      SUPPORTER,
25      /** 使用者 */
26      USER;
27      /** 職名権限の配列 */
28      private static final Auth[] TITLE_AUTHS = new Auth[] { MANAGER, LEADER, MEMBER };
29      /** 職員権限の配列 */
30      private static final Auth[] STAFF_AUTHS = new Auth[] { SUPERVISOR, ADMINISTRATOR, SUPPORTER };
31      /**
32       * {@link #TITLE_AUTHS} の取得
33       * @return {@link #TITLE_AUTHS}
34       */
35      public static Auth[] valuesOfTitle() {
36          return TITLE_AUTHS;
37      }
38      /**
39       * {@link #STAFF_AUTHS} の取得
40       * @return {@link #STAFF_AUTHS}
41       */
42      public static Auth[] valuesOfStaff() {
43          return STAFF_AUTHS;
44      }
45      /**
46       * 職名権限の含有チェック
47       * @return 含有するとき真。含有されないとき偽。
48       */
49      public boolean isTitleAuth() {
50          return Arrays.asList(TITLE_AUTHS).contains(this);
51      }
52      /**
53       * 職員権限の含有チェック
54       * @return 含有するとき真。含有されないとき偽。
55       */
56      public boolean isStaffAuth() {
57          return Arrays.asList(STAFF_AUTHS).contains(this);
58      }
59  }
