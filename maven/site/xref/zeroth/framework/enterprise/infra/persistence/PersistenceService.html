6   package zeroth.framework.enterprise.infra.persistence;
7   import java.io.Serializable;
8   import java.util.Collection;
9   import javax.persistence.EntityManager;
10  import javax.persistence.LockModeType;
11  import javax.persistence.TypedQuery;
12  import zeroth.framework.enterprise.domain.Persistable;
13  import zeroth.framework.standard.shared.Service;
14  /**
15   * 基本データ永続化サービスI/F
16   * @param <T> エンティティ型
17   * @param <ID> 識別子オブジェクト型
18   * @author nilcy
19   */
20  public abstract interface PersistenceService<T extends Persistable<ID>, ID extends Serializable>
21      extends Service {
22      /**
23       * 初期化
24       * <p>
25       * 本サービスのファクトリから実行すること。
26       * </p>
27       * @param clazz エンティティクラス
28       * @param manager エンティティマネージャ
29       */
30      void setup(final Class<T> clazz, EntityManager manager);
31      /**
32       * エンティティの登録
33       * @param entity エンティティ
34       */
35      void persist(T entity);
36      /**
37       * エンティティのID検索
38       * @param id 識別子
39       * @return エンティティ
40       */
41      T find(ID id);
42      /**
43       * エンティティのID検索
44       * @param id 識別子
45       * @param lockModeType ロックモードタイプ
46       * @return エンティティ
47       */
48      T find(Long id, LockModeType lockModeType);
49      /**
50       * エンティティの変更
51       * @param entity 変更後のエンティティ
52       */
53      void merge(T entity);
54      /**
55       * エンティティの削除
56       * @param entity エンティティ
57       */
58      void remove(T entity);
59      /**
60       * エンティティの更新
61       * @param entity エンティティ
62       */
63      void refresh(final T entity);
64      /**
65       * エンティティの更新
66       * @param entity エンティティ
67       * @param lockModeType ロックモードタイプ
68       */
69      void refresh(final T entity, final LockModeType lockModeType);
70      /**
71       * エンティティの保護
72       * @param entity エンティティ
73       * @param lockModeType ロックモードタイプ
74       */
75      void lock(final T entity, LockModeType lockModeType);
76      /** エンティティの同期 */
77      void flush();
78      /**
79       * エンティティの分離
80       * @param entity エンティティ
81       */
82      void detach(T entity);
83      /**
84       * エンティティ含有の確認
85       * @param entity エンティティ
86       * @return 含有するとき真。含有しないとき偽。
87       */
88      boolean contains(final T entity);
89      /**
90       * クエリ結果範囲の設定
91       * @param query クエリ
92       * @param begin 開始位置
93       * @param max 最大件数
94       * @return クエリ
95       */
96      TypedQuery<T> setRange(final TypedQuery<T> query, final int begin, final int max);
97      /**
98       * エンティティのクエリ検索
99       * @param query クエリ
100      * @return 複数エンティティ
101      */
102     Collection<T> findMany(TypedQuery<T> query);
103     /**
104      * エンティティのクエリ検索
105      * @param query クエリ
106      * @return 単一エンティティ
107      */
108     T findOne(TypedQuery<T> query);
109 }
