6   package zeroth.framework.enterprise.shared;
7   import javax.cache.Cache;
8   import javax.cache.CacheBuilder;
9   import javax.cache.CacheManager;
10  import javax.cache.Caching;
11  import javax.cache.transaction.IsolationLevel;
12  import javax.cache.transaction.Mode;
13  /**
14   * キャッシュのユーティリティ
15   * @author nilcy
16   */
17  public final class CacheUtils {
18      /** キャッシュマネージャ */
19      private static final CacheManager MANAGER = Caching.getCacheManager();
20      /** 非公開コンストラクタ */
21      private CacheUtils() {
22      }
23      /**
24       * キャッシュの作成
25       * <p>
26       * 最後に {@link CacheUtils#removeCache(String)} を実行すること。
27       * </p>
28       * @param <K> キーオブジェクト型
29       * @param <V> バリューオブジェクト型
30       * @param cacheName キャッシュ名
31       * @return cache キャッシュ
32       * @see javax.cache.Cache
33       */
34      public static <K, V> Cache<K, V> createCache(final String cacheName) {
35          final CacheBuilder<K, V> builder = MANAGER.createCacheBuilder(cacheName);
36          builder.setStoreByValue(true);
37          builder.setStatisticsEnabled(true);
38          return builder.build();
39      }
40      /**
41       * キャッシュの取得
42       * @param <K> キーオブジェクト型
43       * @param <V> バリューオブジェクト型
44       * @param cacheName キャッシュ名
45       * @return cache キャッシュ
46       * @see javax.cache.Cache
47       */
48      public static <K, V> Cache<K, V> getCache(final String cacheName) {
49          return MANAGER.getCache(cacheName);
50      }
51      /**
52       * デフォルトキャッシュの取得
53       * @param <K> キーオブジェクト型
54       * @param <V> バリューオブジェクト型
55       * @return cache キャッシュ
56       * @see javax.cache.Cache
57       */
58      public static <K, V> Cache<K, V> getCache() {
59          return getCache("application");
60      }
61      /**
62       * キャッシュの削除
63       * @param aCacheName キャッシュ名
64       * @return キャッシュ削除の成否
65       * @see javax.cache.CacheManager#removeCache(String)
66       */
67      public static boolean removeCache(final String aCacheName) {
68          return MANAGER.removeCache(aCacheName);
69      }
70      /**
71       * 読込専用キャッシュの作成
72       * @param <K> キーオブジェクト型
73       * @param <V> バリューオブジェクト型
74       * @param cacheName キャッシュ名
75       * @return cache キャッシュ
76       * @see javax.cache.Cache
77       */
78      public static <K, V> Cache<K, V> createReadCache(final String cacheName) {
79          final CacheBuilder<K, V> builder = MANAGER.createCacheBuilder(cacheName);
80          builder.setStatisticsEnabled(true);
81          builder.setReadThrough(true);
82          builder.setWriteThrough(false);
83          builder.setTransactionEnabled(IsolationLevel.READ_COMMITTED, Mode.LOCAL);
84          return builder.build();
85      }
86  }
