6   package zeroth.framework.standard.domain;
7   /**
8    * 仕様オブジェクト
9    * <p>
10   * 論理演算をするための値オブジェクトである。ビジネスルールを「述語」的に表現するときに使用する。
11   * </p>
12   * @param <T> チェック対象オブジェクト型
13   * @author nilcy
14   */
15  public abstract class AbstractSpecification<T> implements Specification<T> {
16      @Override
17      public abstract boolean isSatisfiedBy(T object);
18      @Override
19      public Specification<T> and(final Specification<T> spec) {
20          return new AndSpecification<>(this, spec);
21      }
22      @Override
23      public Specification<T> or(final Specification<T> spec) {
24          return new OrSpecification<>(this, spec);
25      }
26      @Override
27      public Specification<T> not(final Specification<T> spec) {
28          return new NotSpecification<>(spec);
29      }
30      @Override
31      public Specification<T> xor(final Specification<T> aSpecification) {
32          return new XorSpecification<>(this, aSpecification);
33      }
34      @Override
35      public Specification<T> nand(final Specification<T> aSpecification) {
36          return new NandSpecification<>(this, aSpecification);
37      }
38      @Override
39      public Specification<T> nor(final Specification<T> aSpecification) {
40          return new NorSpecification<>(this, aSpecification);
41      }
42  }
