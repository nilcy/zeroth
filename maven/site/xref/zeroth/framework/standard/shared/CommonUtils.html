6   package zeroth.framework.standard.shared;
7   import java.io.Serializable;
8   import java.lang.reflect.InvocationTargetException;
9   import org.apache.commons.beanutils.BeanUtils;
10  import org.apache.commons.lang3.ObjectUtils;
11  import org.apache.commons.lang3.SerializationUtils;
12  import org.apache.commons.lang3.Validate;
13  /**
14   * レイヤ共通ユーティリティ
15   * @author nilcy
16   */
17  public class CommonUtils {
18      /** コンストラクタ */
19      private CommonUtils() {
20      }
21      /**
22       * 非NULLチェック
23       * @param object チェック対象オブジェクト
24       * @return 非NULLオブジェクト
25       * @throws NullPointerException チェック対象オブジェクトがNULLのとき
26       */
27      public static <T> T notNull(final T object) {
28          return Validate.notNull(object);
29      }
30      /**
31       * NULLセーフ変換
32       * @param object 対象オブジェクト
33       * @param defaultValue NULLセーフオブジェクト
34       * @param <T> 対象オブジェクト型
35       * @return 対象オブジェクトがNULLのときNULLセーフオブジェクト
36       * @see ObjectUtils#defaultIfNull(Object, Object)
37       */
38      public static <T> T nullSafe(final T object, final T defaultValue) {
39          return ObjectUtils.defaultIfNull(object, defaultValue);
40      }
41      /**
42       * プロパティ一括コピー
43       * <p>
44       * オブジェクトの可視性は public でないとコピーできないことに注意すること。
45       * </p>
46       * @param origin コピー元オブジェクト
47       * @param destination コピー先オブジェクト
48       * @throws StandardRuntimeException コピーできないとき
49       */
50      public static void copyProperties(final Object origin, final Object destination) {
51          try {
52              BeanUtils.copyProperties(destination, origin);
53          } catch (final IllegalAccessException e) {
54              throw new StandardRuntimeException(e);
55          } catch (final InvocationTargetException e) {
56              throw new StandardRuntimeException(e);
57          }
58      }
59      /**
60       * プロパティ個別コピー
61       * @param object コピー対象オブジェクト
62       * @param name プロパティ名
63       * @param value プロパティ値
64       * @throws StandardRuntimeException コピーできないとき
65       */
66      public static void copyProperty(final Object object, final String name, final Object value) {
67          try {
68              BeanUtils.copyProperty(object, name, value);
69          } catch (final IllegalAccessException e) {
70              throw new StandardRuntimeException(e);
71          } catch (final InvocationTargetException e) {
72              throw new StandardRuntimeException(e);
73          }
74      }
75      /**
76       * シャローコピー
77       * @param object コピー元オブジェクト
78       * @return シャローコピーしたオブジェクト
79       */
80      public static <T extends Serializable> T shallowCopy(final T object) {
81          return ObjectUtils.clone(object);
82      }
83      /**
84       * ディープコピー
85       * @param object コピー元オブジェクト
86       * @return ディープコピーしたオブジェクト
87       */
88      public static <T extends Serializable> T deepCopy(final T object) {
89          return SerializationUtils.clone(object);
90      }
91  }
