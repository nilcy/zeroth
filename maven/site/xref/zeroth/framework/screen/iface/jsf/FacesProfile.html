34      /** SEPCHR. */

40          super();

49          } catch (final ParseException e) {
50              throw new RuntimeException(e);
51          }
52          final String[] maxResults = profile.getString("MAX_RESULTS").split(SEPCHR);
53          MAX_RESULTS = Integer.parseInt(maxResults[0]);
54          MAX_RESULTS_DL = Integer.parseInt(maxResults[1]);
55          VIEW_NULL = profile.getString("VIEW_NULL");
56          BUNDLE_BASENAME = profile.getString("BUNDLE_BASENAME");
57          BATCH_FLUSH = Boolean.valueOf(profile.getString("BATCH_FLUSH")).booleanValue();
58          CONVERSATION_TIMEOUT = Long.valueOf(profile.getString("CONVERSATION_TIMEOUT")).longValue();
59      }
