26  @RunWith(JeeunitRunner.class)

28  @SuppressWarnings("all")
29  public class QueryPersistenceServiceImplTest {
30      @Inject
31      private QueryPersistenceService<TestExample, Long> testee;
32      @Inject
33      private Logger log;
34      @Test
35      public void testCRUD() {
36          final TestExample e01 = new TestExample("code01");
37          testee.persist(e01);
38          assertThat(e01.getId(), is(1L));
39          assertThat(testee.contains(e01), is(true));
40          final TestExample e01r = testee.find(1L);
41          assertThat(e01r, is(e01));
42          e01r.setCode("code01#1");
43          testee.lock(e01r, LockModeType.PESSIMISTIC_READ);
44          testee.flush();
45          testee.refresh(e01r);
46          testee.refresh(e01r, LockModeType.PESSIMISTIC_WRITE);
47          testee.detach(e01r);
48          e01r.setCode("code01#2");
49          testee.merge(e01r);
50          testee.remove(testee.find(e01r.getId()));
51          assertThat(testee.find(1L, LockModeType.OPTIMISTIC), is(nullValue()));
52      }
53      @Test
54      public void testCriteria() {
55          final CriteriaBuilder b = testee.builder();
56          final CriteriaQuery<TestExample> q = testee.query();
57          final Root<TestExample> r = testee.root();
58          testee.persist(new TestExample("code01"));
59          testee.persist(new TestExample("code02"));
60          final TypedQuery<TestExample> typedQuery = testee
61              .createQuery(q.select(r).where(b.equal(r.get(code), "code01"))
62                  .orderBy(b.asc(r.get(code))).groupBy(r.get(code)));
63          for (final TestExample e : typedQuery.getResultList()) {
64              log.info("TestExampleObject >> " + e.toString());
65          }
66      }
