8   import static org.junit.Assert.*;

15  @SuppressWarnings("all")

17      @Test
18      public void test() {
19          assertThat(CharUtils.CR, is('\r'));
20          assertThat(CharUtils.LF, is('\n'));
21          assertThat(CharUtils.isAscii('a'), is(true));
22          assertThat(CharUtils.isAscii('0'), is(true));
23          assertThat(CharUtils.isAscii('„ÅÇ'), is(false));
24          assertThat(CharUtils.isAsciiAlpha('a'), is(true));
25          assertThat(CharUtils.isAsciiAlpha('0'), is(false));
26          assertThat(CharUtils.isAsciiAlphaLower('a'), is(true));
27          assertThat(CharUtils.isAsciiAlphaLower('A'), is(false));
28          assertThat(CharUtils.isAsciiAlphanumeric('a'), is(true));
29          assertThat(CharUtils.isAsciiAlphanumeric('0'), is(true));
30          assertThat(CharUtils.isAsciiAlphanumeric('-'), is(false));
31          assertThat(CharUtils.isAsciiAlphaUpper('a'), is(false));
32          assertThat(CharUtils.isAsciiAlphaUpper('A'), is(true));
33          assertThat(CharUtils.isAsciiControl('a'), is(false));
34          assertThat(CharUtils.isAsciiControl('\r'), is(true));
35          assertThat(CharUtils.isAsciiNumeric('a'), is(false));
36          assertThat(CharUtils.isAsciiNumeric('0'), is(true));
37          assertThat(CharUtils.isAsciiPrintable('a'), is(true));
38          assertThat(CharUtils.isAsciiPrintable('\r'), is(false));
39          assertThat(CharUtils.toChar("a"), is('a'));
40          assertThat(CharUtils.toChar((String) null, '-'), is('-'));
41          assertThat(CharUtils.toIntValue('0'), is(0));
42          assertThat(CharUtils.toString('a'), is("a"));
43          assertThat(CharUtils.unicodeEscaped(' '), is("\\u0020"));
44      }
