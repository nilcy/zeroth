6   package zeroth.framework.standard.shared.commons.lang;
7   import static org.hamcrest.CoreMatchers.*;
8   import static org.junit.Assert.*;
9   import org.apache.commons.lang3.BooleanUtils;
10  import org.junit.Test;
11  /**
12   * {@link BooleanUtils}
13   * @author nilcy
14   */
15  @SuppressWarnings("all")
16  public class BooleanUtilsTest {
17      @Test
18      public void test() {
19          assertThat(BooleanUtils.and(new boolean[] { true }), is(true));
20          assertThat(BooleanUtils.and(new boolean[] { true, false }), is(false));
21          assertThat(BooleanUtils.isFalse(false), is(true));
22          assertThat(BooleanUtils.isNotFalse(false), is(false));
23          assertThat(BooleanUtils.isNotTrue(true), is(false));
24          assertThat(BooleanUtils.isTrue(true), is(true));
25          assertThat(BooleanUtils.negate(Boolean.FALSE), is(true));
26          assertThat(BooleanUtils.or(new boolean[] { true }), is(true));
27          assertThat(BooleanUtils.or(new boolean[] { true, false }), is(true));
28          assertThat(BooleanUtils.toBoolean(Boolean.TRUE), is(true));
29          assertThat(BooleanUtils.toBoolean(0), is(false));
30          assertThat(BooleanUtils.toBoolean(1), is(true));
31          assertThat(BooleanUtils.toBoolean((String) null), is(false));
32          assertThat(BooleanUtils.toBoolean("true"), is(true));
33          assertThat(BooleanUtils.toBoolean("hoge"), is(false));
34          assertThat(BooleanUtils.toBooleanDefaultIfNull(null, false), is(false));
35          assertThat(BooleanUtils.toBooleanObject(0), is(Boolean.FALSE));
36          assertThat(BooleanUtils.toBooleanObject(1), is(Boolean.TRUE));
37          assertThat(BooleanUtils.toInteger(true), is(1));
38          assertThat(BooleanUtils.toInteger(false), is(0));
39          assertThat(BooleanUtils.toIntegerObject(true), is(Integer.valueOf(1)));
40          assertThat(BooleanUtils.toIntegerObject(false), is(Integer.valueOf(0)));
41          assertThat(BooleanUtils.toString(true, "true", "false"), is("true"));
42          assertThat(BooleanUtils.toString(false, "true", "false"), is("false"));
43          assertThat(BooleanUtils.toStringOnOff(true), is("on"));
44          assertThat(BooleanUtils.toStringOnOff(false), is("off"));
45          assertThat(BooleanUtils.toStringTrueFalse(true), is("true"));
46          assertThat(BooleanUtils.toStringTrueFalse(false), is("false"));
47          assertThat(BooleanUtils.toStringYesNo(true), is("yes"));
48          assertThat(BooleanUtils.toStringYesNo(false), is("no"));
49          assertThat(BooleanUtils.xor(new boolean[] { true }), is(true));
50          assertThat(BooleanUtils.xor(new boolean[] { true, false }), is(true));
51          assertThat(BooleanUtils.xor(new boolean[] { true, true }), is(false));
52      }
53  }
