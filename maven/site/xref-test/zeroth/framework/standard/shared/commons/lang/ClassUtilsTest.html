6   package zeroth.framework.standard.shared.commons.lang;
7   import static org.hamcrest.CoreMatchers.*;
8   import static org.junit.Assert.*;
9   import java.util.ArrayList;
10  import java.util.HashMap;
11  import java.util.List;
12  import org.apache.commons.lang3.ClassUtils;
13  import org.junit.Test;
14  /**
15   * {@link ClassUtils}
16   * @author nilcy
17   */
18  @SuppressWarnings("all")
19  public class ClassUtilsTest {
20      @Test
21      public void test() throws ClassNotFoundException, SecurityException, NoSuchMethodException {
22          List<Class<?>> classes = new ArrayList<Class<?>>() {
23              {
24                  add(Integer.class);
25                  add(Long.class);
26              }
27          };
28          assertThat(ClassUtils.convertClassesToClassNames(classes).toArray(), is(new Object[] {
29              "java.lang.Integer", "java.lang.Long" }));
30          List<String> classNames = new ArrayList<String>() {
31              {
32                  add("java.lang.Integer");
33                  add("java.lang.Long");
34              }
35          };
36          assertThat(ClassUtils.convertClassNamesToClasses(classNames).toArray(),
37              is(classes.toArray()));
38          assertThat(ClassUtils.getAllInterfaces(String.class).toArray(), is(new Object[] {
39              java.io.Serializable.class, java.lang.Comparable.class, java.lang.CharSequence.class }));
40          assertThat(ClassUtils.getAllSuperclasses(HashMap.class).toArray(), is(new Object[] {
41              java.util.AbstractMap.class, java.lang.Object.class }));
42          assertThat(ClassUtils.getClass("java.lang.String").toString(), is("class java.lang.String"));
43          assertThat(ClassUtils.getPackageCanonicalName(String.class), is("java.lang"));
44          assertThat(ClassUtils.getPackageName(String.class), is("java.lang"));
45          assertThat(ClassUtils.getPublicMethod(String.class, "length", new Class[] {}),
46              is(not(nullValue())));
47          assertThat(ClassUtils.getShortCanonicalName(String.class), is("String"));
48          assertThat(ClassUtils.getShortClassName(String.class), is("String"));
49          assertThat(ClassUtils.getSimpleName(String.class), is("String"));
50          assertThat(ClassUtils.isAssignable(String.class, Object.class), is(true));
51          assertThat(ClassUtils.isInnerClass(Foo.class), is(true));
52          assertThat(ClassUtils.isInnerClass(String.class), is(false));
53          assertThat(ClassUtils.isPrimitiveOrWrapper(Integer.class), is(true));
54          assertThat(ClassUtils.isPrimitiveOrWrapper(String.class), is(false));
55          assertThat(ClassUtils.isPrimitiveWrapper(Integer.class), is(true));
56          assertThat(ClassUtils.isPrimitiveWrapper(String.class), is(false));
57          assertThat(ClassUtils.primitivesToWrappers(new Class[] { Integer.class, Long.class }),
58              is(not(nullValue())));
59          assertThat(ClassUtils.primitiveToWrapper(Integer.class), is(not(nullValue())));
60          assertThat(ClassUtils.toClass(1L, 2L), is(not(nullValue())));
61          assertThat(ClassUtils.wrappersToPrimitives(new Class[] { Integer.class, Long.class }),
62              is(not(nullValue())));
63          assertThat(ClassUtils.wrapperToPrimitive(Integer.class), is(not(nullValue())));
64      }

66      }
