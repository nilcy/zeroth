6   package zeroth.framework.standard.shared;
7   import static org.junit.Assert.*;
8   import java.text.MessageFormat;
9   import java.util.logging.Logger;
10  import org.apache.commons.codec.binary.Hex;
11  import org.junit.Test;
12  /**
13   * {@link SecurityUtils} のユニットテスト
14   * @author nilcy
15   */
16  @SuppressWarnings("all")
17  public class SecurityUtilsTest {
18      private static final Logger LOG = Logger.getAnonymousLogger();
19      @Test
20      public void testCreateKey() {
21          final String key01 = SecurityUtils.createKey();
22          final String key02 = SecurityUtils.createKey();
23          assertTrue(key01 != key02);
24          LOG.info(MessageFormat.format("共通鍵01={0},共通鍵02={1}", key01, key02));
25      }
26      @Test
27      public void testEncryptDecrypt() {
28          final String key = SecurityUtils.createKey();
29          final String plainText = "極秘指令書その壱だお";
30          final byte[] rawBytes = plainText.getBytes();
31          final byte[] encBytes = SecurityUtils.encrypt(rawBytes, key);
32          final byte[] decBytes = SecurityUtils.decrypt(encBytes, key);
33          LOG.info(MessageFormat.format(
34              "raw hex={0},encrypted hex={1},decrypted string={2}",
35              new Object[] { Hex.encodeHexString(rawBytes), Hex.encodeHexString(encBytes),
36                  Hex.encodeHexString(decBytes) }));
37          assertTrue(Hex.encodeHexString(rawBytes) != Hex.encodeHexString(encBytes));
38          assertTrue(Hex.encodeHexString(encBytes) != Hex.encodeHexString(decBytes));
39          assertEquals(Hex.encodeHexString(rawBytes), Hex.encodeHexString(decBytes));
40          LOG.info(MessageFormat.format("raw string={0},decrypted string={1}", new String(rawBytes),
41              new String(decBytes)));
42          assertEquals(new String(rawBytes), new String(decBytes));
43      }
44  }
