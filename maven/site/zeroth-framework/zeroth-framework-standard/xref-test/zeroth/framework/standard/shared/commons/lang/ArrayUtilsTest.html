6   package zeroth.framework.standard.shared.commons.lang;
7   import static org.hamcrest.CoreMatchers.*;
8   import static org.junit.Assert.*;
9   import org.apache.commons.lang3.ArrayUtils;
10  import org.junit.Test;
11  /**
12   * {@link ArrayUtils}
13   * @author nilcy
14   */
15  @SuppressWarnings("all")
16  public class ArrayUtilsTest {
17      @Test
18      public void test() {
19          assertThat(ArrayUtils.EMPTY_BOOLEAN_ARRAY.length, is(0));
20          assertThat(ArrayUtils.EMPTY_BOOLEAN_OBJECT_ARRAY, is(new Boolean[] {}));
21          boolean[] testee = ArrayUtils.add(null, true);
22          assertThat(ArrayUtils.clone(testee), is(testee));
23          assertThat(ArrayUtils.contains(testee, true), is(true));
24          assertThat(ArrayUtils.contains(testee, false), is(false));
25          assertThat(ArrayUtils.getLength(testee), is(1));
26          assertThat(ArrayUtils.hashCode(testee), is(not(nullValue())));
27          assertThat(ArrayUtils.indexOf(testee, true), is(not(ArrayUtils.INDEX_NOT_FOUND)));
28          assertThat(ArrayUtils.indexOf(testee, false), is(ArrayUtils.INDEX_NOT_FOUND));
29          assertThat(ArrayUtils.isEmpty(testee), is(false));
30          assertThat(ArrayUtils.isEmpty(ArrayUtils.EMPTY_BOOLEAN_ARRAY), is(true));
31          assertThat(ArrayUtils.isEquals(testee, new boolean[] { true }), is(true));
32          assertThat(ArrayUtils.isNotEmpty(testee), is(true));
33          assertThat(ArrayUtils.isSameLength(testee, new boolean[] { true }), is(true));
34          assertThat(ArrayUtils.isSameType(testee, new boolean[] { true }), is(true));
35          assertThat(ArrayUtils.lastIndexOf(testee, true), is(not(ArrayUtils.INDEX_NOT_FOUND)));
36          assertThat(ArrayUtils.lastIndexOf(testee, false), is(ArrayUtils.INDEX_NOT_FOUND));
37          boolean[] booleanNull = null;
38          assertThat(ArrayUtils.nullToEmpty(booleanNull), is(ArrayUtils.EMPTY_BOOLEAN_ARRAY));
39          assertThat(ArrayUtils.remove(testee, 0), is(ArrayUtils.EMPTY_BOOLEAN_ARRAY));
40          assertThat(ArrayUtils.removeAll(testee, 0), is(ArrayUtils.EMPTY_BOOLEAN_ARRAY));
41          assertThat(ArrayUtils.removeElement(testee, true), is(ArrayUtils.EMPTY_BOOLEAN_ARRAY));
42          assertThat(ArrayUtils.removeElement(testee, false), is(testee));
43          assertThat(ArrayUtils.removeElements(testee, true, false), is(ArrayUtils.EMPTY_BOOLEAN_ARRAY));
44      }
45  }
