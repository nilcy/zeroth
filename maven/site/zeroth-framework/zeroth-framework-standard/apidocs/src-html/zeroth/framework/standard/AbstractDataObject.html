006package zeroth.framework.standard;
007import static org.apache.commons.lang3.builder.ToStringStyle.*;
008import java.io.Serializable;
009import org.apache.commons.lang3.builder.CompareToBuilder;
010import org.apache.commons.lang3.builder.EqualsBuilder;
011import org.apache.commons.lang3.builder.HashCodeBuilder;
012import org.apache.commons.lang3.builder.ToStringBuilder;
013/**
014 * 汎用データオブジェクト
015 * <p>
016 * データオブジェクトの基本的な機能を提供する。 参照オブジェクトと値オブジェクトは本クラスを継承すること。
017 * </p>
018 * @param <T> データオブジェクト型
019 * @author nilcy
020 */
021public abstract class AbstractDataObject<T extends AbstractDataObject<T>> implements Serializable,
022    Comparable<T> {
023    /** 識別番号 */
024    private static final long serialVersionUID = -4438164461402375117L;
025    /** Constructor. */
026    public AbstractDataObject() {
027    }
028    /**
029     * オブジェクトの文字列表現
030     * <p>
031     * インスタンスをもとに短縮形式で自動生成する。
032     * </p>
033     * @return オブジェクトの文字列表現
034     */
035    @Override
036    public String toString() {
037        return ToStringBuilder.reflectionToString(this, SHORT_PREFIX_STYLE);
038    }
039    /** {@inheritDoc} */
040    @Override
041    public boolean equals(final Object aObj) {
042        return EqualsBuilder.reflectionEquals(this, aObj, true);
043    }
044    /** {@inheritDoc} */
045    @Override
046    public int hashCode() {
047        return HashCodeBuilder.reflectionHashCode(this, true);
048    }
049    /** {@inheritDoc} */
050    @Override
051    public int compareTo(final T aObj) {
052        return CompareToBuilder.reflectionCompare(this, aObj, true);
053    }
054}
