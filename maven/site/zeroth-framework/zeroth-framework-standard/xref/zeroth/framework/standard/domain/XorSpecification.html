6   package zeroth.framework.standard.domain;
7   /**
8    * 排他的論理和(XOR)仕様オブジェクト
9    * @param <T> チェック対象オブジェクト型
10   * @author nilcy
11   */
12  public class XorSpecification<T> extends AbstractSpecification<T> {
13      /** 仕様オブジェクト#1 */
14      private final Specification<T> spec1;
15      /** 仕様オブジェクト#2 */
16      private final Specification<T> spec2;
17      /**
18       * 排他的論理和(XOR)仕様のコンストラクタ
19       * @param spec1 仕様オブジェクト#1
20       * @param spec2 仕様オブジェクト#2
21       */
22      public XorSpecification(final Specification<T> spec1, final Specification<T> spec2) {
23          this.spec1 = spec1;
24          this.spec2 = spec2;
25      }
26      /** {@inheritDoc} */
27      @Override
28      public boolean isSatisfiedBy(final T object) {
29          return (this.spec1.isSatisfiedBy(object) && !this.spec2.isSatisfiedBy(object))
30              || (!this.spec1.isSatisfiedBy(object) && this.spec2.isSatisfiedBy(object));
31      }
32  }
