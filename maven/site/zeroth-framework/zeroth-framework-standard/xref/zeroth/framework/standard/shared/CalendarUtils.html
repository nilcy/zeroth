6   package zeroth.framework.standard.shared;
7   import java.util.Date;
8   import org.apache.commons.lang3.time.DateUtils;
9   /**
10   * カレンダーのユーティリティ
11   * @author nilcy
12   */
13  public final class CalendarUtils {
14      /** １つ前の移動量 */
15      private static final int PREV_AMOUNT = -1;
16      /** １つ先の移動量 */
17      private static final int NEXT_AMOUNT = 1;
18      /** 非公開コンストラクタ */
19      private CalendarUtils() {
20      }
21      /**
22       * 先週日付の取得
23       * @param aDate 基点の日付
24       * @return 先週日付
25       */
26      public static Date prevWeek(final Date aDate) {
27          return DateUtils.addWeeks(aDate, PREV_AMOUNT);
28      }
29      /**
30       * 翌週日付の取得
31       * @param aDate 基点の日付
32       * @return 翌週日付
33       */
34      public static Date nextWeek(final Date aDate) {
35          return DateUtils.addWeeks(aDate, NEXT_AMOUNT);
36      }
37      /**
38       * 先月日付の取得
39       * @param aDate 基点の日付
40       * @return 先月日付
41       */
42      public static Date prevMonth(final Date aDate) {
43          return DateUtils.addMonths(aDate, PREV_AMOUNT);
44      }
45      /**
46       * 翌月日付の取得
47       * @param aDate 基点の日付
48       * @return 翌月
49       */
50      public static Date nextMonth(final Date aDate) {
51          return DateUtils.addMonths(aDate, NEXT_AMOUNT);
52      }
53      /**
54       * 前年日付の取得
55       * @param aDate 基点の日付
56       * @return 先月日付
57       */
58      public static Date prevYear(final Date aDate) {
59          return DateUtils.addYears(aDate, PREV_AMOUNT);
60      }
61      /**
62       * 翌年日付の取得
63       * @param aDate 基点の日付
64       * @return 翌月
65       */
66      public static Date nextYear(final Date aDate) {
67          return DateUtils.addYears(aDate, NEXT_AMOUNT);
68      }
69  }
