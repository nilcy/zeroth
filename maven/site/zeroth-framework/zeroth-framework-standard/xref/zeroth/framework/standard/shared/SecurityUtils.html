6   package zeroth.framework.standard.shared;
7   import java.security.InvalidKeyException;
8   import java.security.NoSuchAlgorithmException;
9   import java.security.SecureRandom;
10  import javax.crypto.BadPaddingException;
11  import javax.crypto.Cipher;
12  import javax.crypto.IllegalBlockSizeException;
13  import javax.crypto.KeyGenerator;
14  import javax.crypto.NoSuchPaddingException;
15  import javax.crypto.spec.SecretKeySpec;
16  import org.apache.commons.codec.binary.Base64;
17  /**
18   * セキュリティユーティリティ
19   * <dl>
20   * <dt>事前条件</dt>
21   * <dd>AES256bit鍵を利用するには、JCE管轄ポリシーファイル（無制限強度）をダウンロードして、
22   * %JAVA_HOME%/jre/lib/securityへUS_export_policy.jarとlocal_policy.jar上書きすること。</dd>
23   * </dl>
24   * @author nilcy
25   */
26  public class SecurityUtils {
27      /** 暗号方式名 */
28      private static final String ALGO_NAME = "AES";
29      /** 乱数生成名 */
30      private static final String RAND_NAME = "SHA1PRNG";
31      /** 鍵長(標準128,JCE無制限ポリシー適用256) TODO 256bit鍵 */
32      private static final int KEY_LENGTH = 128;
33      /** 非公開コンストラクタ */
34      private SecurityUtils() {
35      }
36      /**
37       * 共通鍵の作成
38       * <ul>
39       * <li>AES256bit鍵を作成します。</li>
40       * <li>SHA1PRNGで乱数生成したのち、URL安全なBASE64文字列に変換します。</li>
41       * </ul>
42       * @return 共通鍵(URL安全なBASE64文字列)
43       */
44      public static String createKey() {
45          try {
46              final KeyGenerator generator = KeyGenerator.getInstance(ALGO_NAME);
47              generator.init(KEY_LENGTH, SecureRandom.getInstance(RAND_NAME));
48              return Base64.encodeBase64URLSafeString(generator.generateKey().getEncoded());
49          } catch (final NoSuchAlgorithmException e) {
50              throw new IllegalArgumentException(e);
51          }
52      }
53      /**
54       * 暗号
55       * @param plainBytes 平文バイト列
56       * @param base64Key 共通鍵
57       * @return 暗号バイト列
58       */
59      public static byte[] encrypt(final byte[] plainBytes, final String base64Key) {
60          return exec(plainBytes, base64Key, Cipher.ENCRYPT_MODE);
61      }
62      /**
63       * 復号
64       * @param encryptedBytes 暗号バイト列
65       * @param base64Key 共通鍵
66       * @return 平文バイト列
67       */
68      public static byte[] decrypt(final byte[] encryptedBytes, final String base64Key) {
69          return exec(encryptedBytes, base64Key, Cipher.DECRYPT_MODE);
70      }
71      /**
72       * 暗号/復号実行
73       * @param encryptedBytes 入力バイト列(平文/暗号)
74       * @param base64Key 共通鍵
75       * @param cipherMode 暗号/復号モード
76       * @return 出力バイト列(暗号/平文)
77       */
78      private static byte[] exec(final byte[] encryptedBytes, final String base64Key,
79          final int cipherMode) {
80          try {
81              final Cipher cipher = Cipher.getInstance(ALGO_NAME);
82              cipher.init(cipherMode, new SecretKeySpec(Base64.decodeBase64(base64Key), ALGO_NAME));
83              return cipher.doFinal(encryptedBytes);
84          } catch (final InvalidKeyException e) {
85              e.printStackTrace();
86              throw new StandardRuntimeException(e);
87          } catch (final NoSuchAlgorithmException e) {
88              throw new StandardRuntimeException(e);
89          } catch (final NoSuchPaddingException e) {
90              throw new StandardRuntimeException(e);
91          } catch (final IllegalBlockSizeException e) {
92              throw new StandardRuntimeException(e);
93          } catch (final BadPaddingException e) {
94              throw new StandardRuntimeException(e);
95          }
96      }
97  }
