6   package zeroth.framework.service.app;
7   import java.util.logging.Logger;
8   import javax.ws.rs.core.Response.Status;
9   import zeroth.framework.standard.app.GatewayHeader;
10  import zeroth.framework.standard.app.GatewayModel;
11  import zeroth.framework.standard.app.GatewayUtil;
12  import zeroth.framework.standard.shared.StandardException;
13  /**
14   * ゲートウェイサービスのクライアント
15   * @author nilcy
16   */
17  public final class GatewayClient {
18      /** ロガー */
19      private static final Logger LOG = Logger.getGlobal();
20      /** 非公開コンストラクタ */
21      private GatewayClient() {
22      }
23      /**
24       * RPCサービスの実行
25       * @param <T> ボディオブジェクト型
26       * @param aHeader {@link GatewayModel#setHeader(GatewayHeader)}
27       * @param aBody {@link GatewayModel#setBody(String)}
28       * @return {@link GatewayModel}
29       * @throws StandardException ステータスコードがOKでないとき
30       */
31      public static <T> GatewayModel submit(final GatewayHeader aHeader, final T aBody)
32          throws StandardException {
33          final GatewayService gateway = GatewayUtil.getProxy(GatewayService.class);
34          final GatewayModel request = GatewayFactory.create(aBody);
35          request.setHeader(aHeader);
36          final GatewayModel returnModel = gateway.submit(request);
37          if (returnModel.getStatus().getCode().intValue() != Status.OK.getStatusCode()) {
38              LOG.info("returnModel is " + returnModel);
39              throw new StandardException("", returnModel.getStatus().getMessage());
40          }
41          return returnModel;
42      }
43  }
