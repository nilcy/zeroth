6   package zeroth.framework.enterprise.infra.persistence;
7   import javax.persistence.EntityManager;

10  import zeroth.framework.enterprise.shared.Service;
11  /**
12   * 基本データ永続化サービスI/F

18      /**
19       * 初期化
20       * @param aClass 参照オブジェクトクラス
21       * @param aManager 参照オブジェクトマネージャ

27       */
28      void create(T aReferenceObject);
29      /**
30       * 参照オブジェクトの閲覧

34      T read(ID aId);
35      /**
36       * 参照オブジェクトの閲覧
37       * @param aId 識別子
38       * @param aLockModeType ロックモードタイプ

40       */
41      T read(Long aId, LockModeType aLockModeType);

46      void update(T aReferenceObject);

51      void delete(T aReferenceObject);

56      void refresh(final T aReferenceObject);

58       * 参照オブジェクトの更新
59       * @param aReferenceObject 参照オブジェクト

62      void refresh(final T aReferenceObject, final LockModeType aLockModeType);

64       * 参照オブジェクトの保護
65       * @param aReferenceObject 参照オブジェクト

67       */
68      void lock(final T aReferenceObject, LockModeType aLockModeType);
69      /** 参照オブジェクトの同期 */
70      void flush();

73       * @param aReferenceObject 参照オブジェクト

75      void detach(T aReferenceObject);
76      /**

78       * @param aReferenceObject 参照オブジェクト
79       * @return 含有するとき真。含有しないとき偽。

81      boolean contains(final T aReferenceObject);
82  }
