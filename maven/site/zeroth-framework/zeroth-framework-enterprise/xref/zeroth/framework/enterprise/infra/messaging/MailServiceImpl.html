6   package zeroth.framework.enterprise.infra.messaging;
7   import javax.annotation.Resource;
8   import javax.ejb.Stateless;
9   import javax.enterprise.inject.Alternative;
10  import javax.mail.Message;
11  import javax.mail.MessagingException;
12  import javax.mail.Multipart;
13  import javax.mail.Session;
14  import javax.mail.Transport;
15  import javax.mail.internet.AddressException;
16  import javax.mail.internet.InternetAddress;
17  import javax.mail.internet.MimeMessage;
18  import org.apache.commons.lang3.Validate;
19  import zeroth.framework.enterprise.shared.EnterpriseException;
20  /**
21   * メールサービスI/F
22   * @author nilcy
23   */
24  @Stateless
25  @Alternative
26  public class MailServiceImpl implements MailService {
27      /** 識別番号 */
28      private static final long serialVersionUID = 5415112327668280272L;
29      /** メールセッション */
30      @Resource(name = "mail/primary")
31      private Session session;
32      @Override
33      public void setup(final Session aSession) {
34          this.session = aSession;
35      }
36      @Override
37      public void send(final String aFrom, final String aTo, final String aSubject,
38          final Object aObjectBody, final String aContentType) throws EnterpriseException {
39          try {
40              final Message message = createMessage(aFrom, aTo, aSubject);
41              message.setContent(aObjectBody, aContentType);
42              send(message);
43          } catch (final MessagingException e) {
44              throw new EnterpriseException(e);
45          }
46      }
47      @Override
48      public void send(final String aFrom, final String aTo, final String aSubject,
49          final String aTextBody) throws EnterpriseException {
50          try {
51              final Message message = createMessage(aFrom, aTo, aSubject);
52              message.setContent(aTextBody, "text/plain");
53              send(message);
54          } catch (final MessagingException e) {
55              throw new EnterpriseException(e);
56          }
57      }
58      @Override
59      public void send(final String aFrom, final String aTo, final String aSubject,
60          final Multipart aMultipartBody) throws EnterpriseException {
61          try {
62              final Message message = createMessage(aFrom, aTo, aSubject);
63              message.setContent(aMultipartBody);
64              send(message);
65          } catch (final MessagingException e) {
66              throw new EnterpriseException(e);
67          }
68      }
69      /**
70       * メッセージの作成
71       * @param aFrom FROMアドレス
72       * @param aTo TOアドレス
73       * @param aSubject メール件名
74       * @return メッセージ
75       * @throws EnterpriseException 指定アドレス、メッセージの例外
76       */
77      private Message createMessage(final String aFrom, final String aTo, final String aSubject)
78          throws EnterpriseException {
79          Validate.notNull(this.session, "メールセッションはないといけません。");
80          try {
81              final Message message = new MimeMessage(this.session);
82              message.setFrom(new InternetAddress(aFrom));
83              message.addRecipient(Message.RecipientType.TO, new InternetAddress(aTo));
84              message.setSubject(aSubject);
85              return message;
86          } catch (final AddressException e) {
87              throw new EnterpriseException(e);
88          } catch (final MessagingException e) {
89              throw new EnterpriseException(e);
90          }
91      }
92      /**
93       * メール送信
94       * @param aMessage メッセージ
95       * @throws EnterpriseException メッセージの例外
96       */
97      private static void send(final Message aMessage) throws EnterpriseException {
98          try {
99              Transport.send(aMessage);
100         } catch (final MessagingException e) {
101             throw new EnterpriseException(e);
102         }
103     }
104 }
